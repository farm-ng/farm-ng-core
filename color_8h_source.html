49  static Color constexpr from(
50  ColorHue hue,
51  ColorBrightness brightness = ColorBrightness::medium,
52  ColorSaturation sat = ColorSaturation::normal,
53  float a = 1.0) {
54  float factor = 1.f;
55  float zero = 0.f;
56  float half = 0.5f;
57  switch (brightness) {
58  case ColorBrightness::bright: {
59  break;
60  }
61  case ColorBrightness::medium: {
62  factor = 0.75f;
63  break;
64  }
65  case ColorBrightness::dark: {
66  factor = 0.5f;
67  break;
68  }
69  }
70  switch (sat) {
71  case ColorSaturation::normal: {
72  break;
73  }
74  case ColorSaturation::neon: {
75  zero = 0.5f;
76  half = 0.75f;
77  break;
78  }
79  }
80  switch (hue) {
81  case ColorHue::red: {
82  return Color(1.0, zero, zero, a) * factor;
83  break;
84  }
85  case ColorHue::orange: {
86  return Color(1.0, half, zero, a) * factor;
87  break;
88  }
89  case ColorHue::yellow: {
90  return Color(1.0, 1.0, zero, a) * factor;
91  break;
92  }
93  case ColorHue::chartreuse: {
94  return Color(half, 1.0, zero, a) * factor;
95  break;
96  }
97  case ColorHue::green: {
98  return Color(zero, 1.0, zero, a) * factor;
99  break;
100  }
101  case ColorHue::turquoise: {
102  return Color(zero, 1.0, half, a) * factor;
103  break;
104  }
105  case ColorHue::cyan: {
106  return Color(zero, 1.0, 1.0, a) * factor;
107  break;
108  }
109  case ColorHue::azure: {
110  return Color(zero, half, 1.0, a) * factor;
111  break;
112  }
113  case ColorHue::blue: {
114  return Color(zero, zero, 1.0, a) * factor;
115  break;
116  }
117  case ColorHue::violet: {
118  return Color(half, zero, 1.0, a) * factor;
119  break;
120  }
121  case ColorHue::magenta: {
122  return Color(1.0, zero, 1.0, a) * factor;
123  break;
124  }
125  case ColorHue::pink: {
126  return Color(1.0, zero, half, a) * factor;
127  break;
128  }
129  }
130  FARM_IMPL_ABORT();
131  }
